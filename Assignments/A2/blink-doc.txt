#include "msp.h"#define FREQ_15_MHZ 1500000#define FREQ_3_MHZ  3000000#define FREQ_6_MHZ  6000000#define FREQ_12_MHZ 12000000#define FREQ_24_MHZ 24000000#define FREQ_48_MHZ 48000000/** * main.c */void main(void){    int i;       WDT_A->CTL = WDT_A_CTL_PW | WDT_A_CTL_HOLD;    // Stop WDT       // P1.0 set as GPIO       P1->SEL0 &= ~BIT0;               // Clear bit 0 of the P1->SEL0 register       P1->SEL1 &= ~BIT0;               // Clear bit 0 of the P1->SEL1 register       P1->DIR |= BIT0;                 // P1.0 set as output       while (1)                        // continuous loop       {           P1->OUT ^= BIT0;                // Toggle P1.0 LED           delay_ms(1, FREQ_15_MHZ);                  }}void delay_ms(int delay, int freq){    int i;    for(i = (delay / 1000) * freq; i > 0; i--);}void delay_us(int delay, int freq){    int i;    for(i = (delay / 1000000) * freq; i > 0; i--);}void set_DCO(int freq){    CS->KEY = 0X0000695A;    switch(int freq)        {            case FREQ_1_5_MHz:                CS->CTL0 = CS_CTL0_DCORSEL_0; // set DCO = 1.5 MHz                break;            case FREQ_3_MHz:                CS->CTL0 = CS_CTL0_DCORSEL_1; // set DCO = 3 MHz                break;            case FREQ_6_MHz:                CS->CTL0 = CS_CTL0_DCORSEL_2; // set DCO = 6 MHz                break;            case FREQ_12_MHz:                CS->CTL0 = CS_CTL0_DCORSEL_3; // set DCO = 12 MHz                break;            case FREQ_24_MHz:                CS->CTL0 = CS_CTL0_DCORSEL_4; // set DCO = 24 MHz                break;            case FREQ_48_MHz:                CS->CTL0 = CS_CTL0_DCORSEL_5; // set DCO = 48 MHz                break;        }    CS -> CTL1 = CS_CTL1_SELA_2 | CS_CTL1_SELS_3 | CS_CTL1_SELM_3;    CS->KEY = 0;}
